src/render/musicPicker.ts:

  export type MusicPickStrategy = 'trim-fade' | 'reroll' | 'crossfade';

  export interface MusicPick {
    track: string;          // pool entry file name (e.g. 'atmos-3.mp3')
    durationSec: number;    // pool entry duration (NOT episode duration)
    strategy: MusicPickStrategy;
    reroll: number;         // 0 for first pick; N for the (N+1)-th attempt
  }

  export class MusicPickError extends Error {
    constructor(input: { reason: string; remediation: string });
  }

  export const MUSIC_REROLL_LIMIT = 32; // soft upper bound — pool size is the
                                        // hard ceiling, but guard against
                                        // pathologically large pools.

  export function pickTrack(opts: {
    manifestHash: string;            // 'sha256:<hex>' from runPrepare
    totalDurationSec: number;        // episode runtime budget
    pool: MusicPoolEntry[];
  }): MusicPick;

  // Algorithm (pinned by tests — do not deviate):
  //
  // 1. If pool.length === 0: throw MusicPickError({reason: 'music pool is empty', remediation: 'commit at least one .mp3 + entry in remotion/assets/music/index.json'})
  //
  // 2. Compute the initial index from the seed:
  //      seed0 = sha256(manifestHash + ':roll:0').slice(0, 16)        // 64-bit hex
  //      idx0  = Number(BigInt('0x' + seed0) % BigInt(pool.length))
  //
  // 3. If pool[idx0].durationSec >= totalDurationSec:
  //      return { track: pool[idx0].file, durationSec: pool[idx0].durationSec, strategy: 'trim-fade', reroll: 0 }
  //
  // 4. Otherwise re-roll deterministically, up to MUSIC_REROLL_LIMIT or pool.length (whichever is smaller — both caps are enforced):
  //      for r in 1..min(pool.length, MUSIC_REROLL_LIMIT):
  //        seedR = sha256(manifestHash + ':roll:' + r).slice(0, 16)
  //        idxR  = Number(BigInt('0x' + seedR) % BigInt(pool.length))
  //        if pool[idxR].durationSec >= totalDurationSec:
  //          return { track: pool[idxR].file, durationSec: pool[idxR].durationSec, strategy: 'reroll', reroll: r }
  //
  // 5. If no track is long enough after exhausting the re-roll budget:
  //      longest = pool sorted by durationSec descending, then by file
  //                ascending (tiebreak — keeps the choice deterministic
  //                when two tracks share the maximum duration)
  //      return { track: longest.file, durationSec: longest.durationSec, strategy: 'crossfade', reroll: <attempt count> }
  //      where <attempt count> = min(pool.length, MUSIC_REROLL_LIMIT)
  //
  // Determinism contract: the function must be pure (only inputs: opts).
  // No reads from process.env, no Date.now(), no Math.random(). Tests
  // verify this by running 10 successive calls and asserting equal results.
