  // Algorithm (pinned — executor must not deviate):

  // 2. Load channel config via loadChannel(manifest.kid, { path: opts.channelsPath }).
  // 3. Load music pool via loadMusicPool({ indexPath: opts.musicIndexPath }).
  // 4. Compute MusicPick via pickTrack({ manifestHash: manifest.manifestHash, totalDurationSec: manifest.totalDurationSec, pool }).
  //    Determine the chosen track's absPath by looking up pool.find(e => e.file === pick.track).absPath.
  // 5. Idempotency check (BEFORE bundling, BEFORE any Remotion work):
  //    - If manifest.render exists AND manifest.render.manifestHash === manifest.manifestHash AND !opts.force:
  //        Verify .golazo/episode.mp4 + .golazo/thumb.png still exist on disk. If both exist:
  //          return { skipped: true, reason: 'hash-match', ... }.
  //        If either is missing: continue to render (treat as recoverable — recorded hash matched but files were deleted).
  //    - If manifest.render exists AND manifest.render.manifestHash !== manifest.manifestHash AND !opts.force:
  //        reason = 'hash-changed'; continue to render.
  //    - If !manifest.render: reason = 'first-render'; continue.
  //    - If opts.force: reason = 'force'; continue.
  // 6. Bundle Remotion: bundle({ entryPoint: 'remotion/Root.tsx', /* default webpackOverride */ }) -> bundleLocation string.
  // 7. Build episodeInputProps from manifest + channel + pick:

  export const ACRONYM_ALLOW_LIST = new Set(['sc', 'fc', 'ac']);

  export function prettyOpponent(slug: string): string;
  // Algorithm:
  //   1. Split on hyphens
  //   2. For each part: if ACRONYM_ALLOW_LIST.has(part.toLowerCase()) -> upper-case it; else title-case (first letter upper, rest lower)
  //   3. Join with space
  // Examples (pinned by tests):
  //   'united'   -> 'United'
  //   'city-sc'  -> 'City SC'
  //   'ac-milan' -> 'AC Milan'
  //   'real-madrid' -> 'Real Madrid'
