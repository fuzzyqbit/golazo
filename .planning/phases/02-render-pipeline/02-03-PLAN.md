  export function computeChapterRhythm(totalClips: number): ChapterRhythm;
  // <= 5 -> 'every-clip'
  // >  5 -> 'every-3'

  export function shouldRenderChapterCardBefore(opts: { clipIndex: number; totalClips: number }): boolean;
  // For 'every-clip': true for every clipIndex (0-based)
  // For 'every-3':   true for clipIndex === 0 || clipIndex % 3 === 0

remotion/composition/timing.ts:

  export type EpisodeSegmentKind = 'title' | 'chapter' | 'clip' | 'outro';

  export interface EpisodeSegment {
    kind: EpisodeSegmentKind;
    startFrame: number;
    durationInFrames: number;
    /** For 'clip' kind only: index into the input.clips array. */
    clipIndex?: number;
    /** For 'chapter' kind only: the clipIndex of the NEXT clip this card heads. */
    nextClipIndex?: number;
  }

  export interface EpisodeTimeline {
    fps: number;
    totalDurationInFrames: number;
    segments: EpisodeSegment[];
  }

  export function computeEpisodeTimeline(input: {
    clips: { durationSec: number }[];
    /** Defaults to MOTION.fps from remotion/theme. */
    fps?: number;
  }): EpisodeTimeline;

  // Frame conversion rule: per-clip durationInFrames = Math.ceil(durationSec * fps).
  // Ceil (not round) so the rendered clip never CUTS OFF — it may include up to
  // one extra blank frame at the end, which the cinematic grade hides.
  //
  // First clip durationInFrames doubles (playbackRate=0.5 means the source plays
  // for 2x its source duration in episode-wall-clock time). Pinned by tests.
  //
  // Segments:
  //   1. TitleCard:  startFrame=0,                                  duration = MOTION.titleCardFrames
  //   2. For each clipIndex c in 0..clips.length-1:
  //      a. If shouldRenderChapterCardBefore({ clipIndex: c, totalClips }): ChapterCard with duration = MOTION.chapterCardFrames
  //      b. Clip with duration = ceil(clips[c].durationSec * fps) * (c === 0 ? 2 : 1)
  //   3. Outro: duration = MOTION.outroFrames

  export interface ClipPlayback {
    playbackRate: number;  // 0.5 for first clip, 1.0 otherwise
    muted: boolean;        // true for first clip (let music carry), false otherwise
  }

  export function getClipPlayback(clipIndex: number): ClipPlayback;
  // clipIndex === 0 -> { playbackRate: 0.5, muted: true }
  // clipIndex >  0  -> { playbackRate: 1.0, muted: false }

  export const MUSIC_VOLUME_MUTED   = 0;     // music muted entirely (first slo-mo clip — REN-04)
  export const MUSIC_VOLUME_DUCKED  = 0.2;   // music ducked under subsequent clip audio
  export const MUSIC_VOLUME_BASELINE = 0.7;  // music at baseline (title / chapter / outro)

  export function musicVolumeAtFrame(
    frame: number,
    timeline: EpisodeTimeline,
    fps?: number,
  ): number;
  // Algorithm (step-function — see Plan SUMMARY for the deferred boundary-ramp polish):
  //   1. Find the segment in timeline.segments whose [startFrame, startFrame+durationInFrames) range contains `frame`.
  //      - Out-of-bounds frame (before frame 0 or past totalDurationInFrames) returns MUSIC_VOLUME_BASELINE.
  //   2. If the segment is 'clip' AND clipIndex === 0 -> return MUSIC_VOLUME_MUTED.
  //   3. If the segment is 'clip' (any other index) -> return MUSIC_VOLUME_DUCKED.
  //   4. Otherwise ('title' | 'chapter' | 'outro') -> return MUSIC_VOLUME_BASELINE.
  //
  // The `fps` parameter is currently UNUSED (step-function has no time-based ramp);
  // it is in the signature so a future polish PR can add a boundary ramp (REN-04
  // smoothing) without changing every call site. Plan 02-03 ships the step-function;
  // ramp is a deferred polish item recorded in the SUMMARY.

    ```
    if input.clips.length === 0: throw new Error('cannot render episode with zero clips')

    const fps = input.fps ?? MOTION.fps;
    const segments: EpisodeSegment[] = [];
    let cursor = 0;

    // Title
    segments.push({ kind: 'title', startFrame: cursor, durationInFrames: MOTION.titleCardFrames });
    cursor += MOTION.titleCardFrames;

    // Per-clip with optional preceding chapter card
    const totalClips = input.clips.length;
    for (let c = 0; c < totalClips; c++) {
      if (shouldRenderChapterCardBefore({ clipIndex: c, totalClips })) {
        segments.push({ kind: 'chapter', startFrame: cursor, durationInFrames: MOTION.chapterCardFrames, nextClipIndex: c });
        cursor += MOTION.chapterCardFrames;
      }
      const baseFrames = Math.ceil(input.clips[c].durationSec * fps);
      const clipFrames = c === 0 ? baseFrames * 2 : baseFrames;
      segments.push({ kind: 'clip', startFrame: cursor, durationInFrames: clipFrames, clipIndex: c });
      cursor += clipFrames;
    }

    // Outro
    segments.push({ kind: 'outro', startFrame: cursor, durationInFrames: MOTION.outroFrames });
    cursor += MOTION.outroFrames;

    return { fps, totalDurationInFrames: cursor, segments };
    ```

    ```ts
    ): number {

      );

      if (seg.kind === 'clip' && seg.clipIndex === 0) return MUSIC_VOLUME_MUTED;
      if (seg.kind === 'clip') return MUSIC_VOLUME_DUCKED;
      return MUSIC_VOLUME_BASELINE;  // 'title' | 'chapter' | 'outro'
    }
    ```
