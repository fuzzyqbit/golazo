<!-- Why better-sqlite3 over node:sqlite (Node 22+ built-in): -->
<!--   1. better-sqlite3 is sync — matches our single-process server's write-light workload -->
<!--      (a single scan inserts ~10s-100s rows; UI queries are read-only). Sync API removes -->
<!--      Promise plumbing throughout the cache/watcher modules. -->
<!--   2. node:sqlite shipped in Node 22 but is still flagged unstable in 22.x docs; -->
<!--      better-sqlite3 has 5+ years of production usage and a richer test surface. -->
<!--   3. better-sqlite3 prepares statements once and binds — fastest read path in Node. -->
<!--   4. Re-evaluate at v2.1 milestone close when node:sqlite is stable. -->

better-sqlite3 API used here (from npmjs / Context7 — confirm at execution time via `cd web && npx ctx7 docs '/wiselibs/better-sqlite3' "prepare statement bind run all transaction"`):
  import Database, { type Database as DB } from 'better-sqlite3';
  const db = new Database(dbPath);                              // sync, creates file if missing
  db.pragma('journal_mode = WAL');                              // recommended for concurrent reads
  db.exec(EPISODES_TABLE_SQL);                                  // sync DDL
  const stmt = db.prepare('INSERT OR REPLACE INTO ... VALUES (?, ?, ...)');
  stmt.run(...args);                                            // sync — returns RunResult { changes, lastInsertRowid }
  const all: unknown[] = db.prepare('SELECT * FROM ...').all(); // sync — returns rows
  const one = db.prepare('SELECT ... WHERE manifest_hash = ?').get(hash); // single row or undefined
  const tx = db.transaction((rows: EpisodeIndex[]) => { ... });
  tx(rows);                                                     // sync — wraps in BEGIN/COMMIT/ROLLBACK
  db.close();                                                   // sync, idempotent

  import type { Database as BetterSqliteDB } from 'better-sqlite3';

  export interface Cache {
    db: BetterSqliteDB;
    dbPath: string;
  }

  export const DEFAULT_CACHE_DB_PATH: string;
  // = path.resolve(process.cwd(), 'web/data/index.db') when invoked from repo root;
  // Plan 04 may override via openCache(opts) on startup to keep tests sandboxed.

  export function openCache(opts?: { dbPath?: string }): Cache;
  // 1. Resolve dbPath (default DEFAULT_CACHE_DB_PATH).
  // 2. mkdirSync(dirname(dbPath), { recursive: true }) — ensures web/data/ exists.
  // 3. new Database(dbPath) — creates file if missing.
  // 4. db.pragma('journal_mode = WAL').
  // 5. db.exec(EPISODES_TABLE_SQL) — idempotent (IF NOT EXISTS).
  // 6. db.exec(EPISODES_INDEX_SQL) — idempotent (IF NOT EXISTS).
  // 7. Return { db, dbPath }.

  export function closeCache(cache: Cache): void;
  // db.close() — idempotent; no-op if already closed.

  export function upsertEpisode(cache: Cache, row: EpisodeIndex): void;
  // INSERT OR REPLACE INTO episodes VALUES (?, ?, ..., ?)
  // Prepared statement cached per Cache instance (lazy memoization on first call).

  export function deleteEpisode(cache: Cache, manifestHash: string): void;
  // DELETE FROM episodes WHERE manifest_hash = ?

  export function queryAllEpisodes(cache: Cache): EpisodeIndex[];
  // SELECT * FROM episodes ORDER BY kid ASC, date DESC, game_folder ASC
  // Rows pass through ROW_TO_EPISODE_INDEX so types are sound.

  export function queryEpisodeByHash(cache: Cache, manifestHash: string): EpisodeIndex | null;
  // SELECT * FROM episodes WHERE manifest_hash = ? LIMIT 1; returns null if no row.

  export function rebuildFromScan(cache: Cache, scanResult: ScanResult): void;
  // Inside a single transaction:
  //   DELETE FROM episodes;
  //   for each row in scanResult.episodes: upsertEpisode(cache, row).
  // Atomic — if any insert throws, the transaction rolls back and the cache is unchanged.

web/src/lib/cacheInvalidation.ts:

  export const TRACKED_FILES: readonly string[];
  // = ['.golazo/manifest.json', '.golazo/episode.mp4', '.golazo/thumb.png', '.golazo/publish.json'];
  // Sourced from scanner.ts constants — no duplication.

  export function peekManifestHashFromFile(absFolderPath: string): string | null;
  // Cheap regex extract from <absFolderPath>/.golazo/manifest.json without full JSON.parse.
  // Regex: /"manifestHash"\s*:\s*"(sha256:[0-9a-f]{64})"/  (top-level pin via the canonical format).
  // Returns the captured hash on success, null on absent file / no match / unreadable file.
  // Rationale (D-19): the invalidation hot-path runs on EVERY chokidar event; a regex over a
  // 4kB manifest is ~30x faster than JSON.parse + zod safeParse. The pessimistic fallback (no
  // match → null → treat as stale → full re-scan) is safe.

  export function trackedFileMtimes(absFolderPath: string): { path: string; mtimeMs: number }[];
  // For each TRACKED_FILES path, statSync(join(absFolderPath, path)).mtimeMs;
  // Missing files contribute mtimeMs = 0 (lower than any cachedRow.scanned_at_ms).

  export function isRowStale(input: {
    cachedRow: EpisodeIndex;
    diskManifestHash: string | null;  // from peekManifestHashFromFile
    trackedMtimes: { mtimeMs: number }[];
  }): boolean;
  // Returns true when:
  //   diskManifestHash !== cachedRow.manifestHash  (hash drift — definitive)
  //   OR any trackedMtimes[i].mtimeMs > cachedRow.scannedAtMs (mtime drift)
  // Returns false otherwise.
  // Pure function — disk access happens in the helpers above.
