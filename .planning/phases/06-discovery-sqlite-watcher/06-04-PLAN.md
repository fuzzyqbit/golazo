  export interface DiscoveryRuntime {
    cache: Cache;
    watcher: WatcherHandle | null;  // null when rootPath does not exist on disk
    rootPath: string;
    dbPath: string;
    warnings: WarningBag;           // from the most recent full scan; Phase 7 will surface this
    rootMissing: boolean;           // true when rootPath does not exist on startup
  }

  export interface DiscoveryRuntimeStatus {
    rootPath: string;
    dbPath: string;
    episodeCount: number;
    warnings: { brokenFolders: number; invalidManifests: number; invalidPublishRecords: number };
    watcherReady: boolean;
    rootMissing: boolean;
  }

  export function resolveGolazoRoot(env?: NodeJS.ProcessEnv): string;
  // Resolution order:
  //   1. env.GOLAZO_ROOT (absolute path expected; resolve to absolute via path.resolve)
  //   2. path.join(os.homedir(), 'golazo')
  // Returns the absolute path.

  export function getDiscoveryRuntime(opts?: { rootPath?: string; dbPath?: string }): Promise<DiscoveryRuntime>;
  // Lazy singleton — first call creates; subsequent calls return the cached instance.
  // opts.rootPath / opts.dbPath are FOR TESTS ONLY — passing them after the singleton is built
  // returns the existing instance unchanged (no re-init). Logs a warning to stderr when opts
  // are provided but ignored. Plan 04 production code does NOT pass opts.
  //
  // Initialization steps:
  //   1. Resolve rootPath via opts.rootPath ?? resolveGolazoRoot().
  //   2. Resolve dbPath via opts.dbPath ?? DEFAULT_CACHE_DB_PATH.
  //   3. cache = openCache({ dbPath })
  //   4. rootMissing = !existsSync(rootPath)
  //   5. If rootMissing: warnings = createWarningBag(); watcher = null;
  //      return { cache, watcher: null, rootPath, dbPath, warnings, rootMissing: true }
  //   6. Else:
  //      a. scanResult = scanGolazoRoot(rootPath)
  //      b. rebuildFromScan(cache, scanResult)
  //      c. watcher = startWatcher({ cache, rootPath })
  //      d. await watcher.ready
  //      e. Return { cache, watcher, rootPath, dbPath, warnings: scanResult.warnings, rootMissing: false }
  //   7. Register a SIGINT + SIGTERM handler that calls shutdownDiscoveryRuntime()
  //      (idempotent — safe under multiple signals)

  export function shutdownDiscoveryRuntime(): Promise<void>;
  // Idempotent. If the singleton is non-null:
  //   1. await watcher?.close()
  //   2. closeCache(cache)
  //   3. Reset the module-level singleton to null so a subsequent
  //      getDiscoveryRuntime() rebuilds (useful for tests).

  export function getDiscoveryRuntimeStatus(): Promise<DiscoveryRuntimeStatus>;
  // Convenience for the debug route — calls getDiscoveryRuntime() then constructs the status object.
