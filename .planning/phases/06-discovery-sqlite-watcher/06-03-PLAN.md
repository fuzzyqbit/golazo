<!-- Why chokidar over node:fs/watch:
     1. node:fs.watch has documented inconsistencies on macOS (events fire on parent dir for child changes, recursive flag is Mac-only and unreliable for deeper trees)
     2. chokidar normalizes these across platforms and offers `awaitWriteFinish` (helpful for atomic-write detection)
     3. chokidar's `ready` event provides a deterministic "initial walk complete" signal we need before reporting startup success in Plan 04
     4. ~3 MB unzipped dep — acceptable for a dev workstation; not shipped to clients (web UI is local-only)
-->

    awaitWriteFinish: { stabilityThreshold: 200, pollInterval: 50 },  // wait for write to settle (atomic-write friendly)
  });
  watcher.on('ready', () => { ... });
  watcher.on('add',     (path) => { ... });
  watcher.on('change',  (path) => { ... });
  watcher.on('unlink',  (path) => { ... });
  watcher.on('unlinkDir', (path) => { ... });
  await watcher.close();

  // The default `ignored: /(^|[/\\])\../` filter excludes ALL dotfiles — including `.golazo/`!
  // Override with a predicate function that allows `.golazo/` and its contents:
  ignored: (p: string) => {
    if (p.includes('node_modules') || p.includes('/.git/')) return true;
    // Allow .golazo/ — strip leading rootPath segment then check segments
    const rel = path.relative(rootPath, p);
    const segments = rel.split(path.sep);
    return segments.some(s => s.startsWith('.') && s !== '.golazo');
  }

    /** Schedule fn(key) to fire after windowMs of quiet for this key. Subsequent triggers within the window restart the timer. */
    trigger(key: string): void;
    /** Force-fire all pending timers immediately (synchronously). Used in close() to drain queued events. */
    flush(): void;
    /** Drop all pending timers without firing. Used in close() when we want to abandon queued events. */
    cancel(): void;
    /** Total count of pending keys (for tests). */
    pendingCount(): number;
  }

    windowMs: number,
  ): PerFolderDebouncer;
