src/publish/publishRecord.ts:

  import { PRIVACY_STATUS } from './uploader.js';

  export const PUBLISH_RECORD_FILE_NAME = '.golazo/publish.json';

  export const publishRecordSchema = z.object({
    videoId: z.string().min(1),
    watchUrl: z.string().regex(/^https:\/\/youtu\.be\//),
    uploadedAt: z.string().datetime(),
    channelId: z.string().regex(/^UC/),
    // Use z.literal('unlisted') for the runtime gate AND a compile-time assertion
    // below to bind the schema literal to the uploader's PRIVACY_STATUS constant.
    // This way any future drift in either side breaks the typecheck before runtime.
    privacyStatus: z.literal('unlisted'),
  });

  // Compile-time bridge between the schema literal and uploader's PRIVACY_STATUS:
  // if PRIVACY_STATUS ever changes from 'unlisted', this assertion fails the build.
  // Rationale for not using z.literal(PRIVACY_STATUS) directly: keeping the bare
  // 'unlisted' string in the zod call surfaces in the verify grep gate
  // `grep -q "z.literal\\('unlisted'\\)" ...`, which is the most legible
  // source-level guarantee. The `satisfies` line preserves the cross-module link.
  type _PrivacyStatusBinding = 'unlisted' satisfies typeof PRIVACY_STATUS;

  export type PublishRecordDoc = z.infer<typeof publishRecordSchema>;
  // Note: identical shape to PublishRecord from uploader.ts. The runner casts/parses freely.

  export function readPublishRecord(folderPath: string): PublishRecordDoc | null;
  // Returns null when <folderPath>/.golazo/publish.json does not exist.
  // Throws PublishError on JSON parse failure or schema validation failure.

  export function writePublishRecord(folderPath: string, record: PublishRecordDoc): void;
  // Creates .golazo/ if missing. JSON.stringify(record, null, 2) + '\n'. utf8.
  // Mirrors writeManifest from Plan 01-05 line-for-line.

src/publish/runner.ts:

  export type PublishReason = 'first-publish' | 'video-exists' | 'force';

  export interface RunPublishOpts {
    folderPath: string;
    channelsPath?: string;        // default './channels.yaml'
    force?: boolean;              // default false
    clientId?: string;            // OAuth client id; default env GOOGLE_CLIENT_ID
    clientSecret?: string;        // default env GOOGLE_CLIENT_SECRET
    /**
     * Optional retry-policy override for tests (default = production [1000, 4000, 16000]).
     * Forwarded to publishWithRetry.
     */
    retryOpts?: WithRetryOpts;
    /** Inject clock for deterministic uploadedAt in tests. */
    clock?: () => Date;
  }

  export interface RunPublishResult {
    skipped: boolean;
    reason: PublishReason;
    publishRecordPath: string;    // absolute
    record: PublishRecordDoc;
  }

  export async function runPublish(opts: RunPublishOpts): Promise<RunPublishResult>;
  // Algorithm (16 steps mirroring Plan 02-04's runRender):
  //
  //  1. Resolve absFolder = resolve(opts.folderPath).
  //  2. Read manifest via readManifest(absFolder). Null → throw PublishError({ field: 'manifestPath', reason: 'manifest not found...', remediation: "run 'golazo prepare <folder>' first" }).
  //  3. Verify .golazo/episode.mp4 exists. Missing → throw PublishError({ field: 'episodePath', reason: 'episode.mp4 not found in .golazo/', remediation: "run 'golazo render <folder>' first" }).
  //  4. Verify .golazo/thumb.png exists. Missing → throw PublishError({ field: 'thumbnailPath', reason: 'thumb.png not found in .golazo/', remediation: "run 'golazo render <folder>' first" }).
  //  5. Load channel via loadChannel(manifest.kid, { path: opts.channelsPath }). Token-exists check is enforced by the default loader; missing token → ChannelsConfigError with `run 'golazo auth <kid>'` remediation (propagates).
  //  6. Idempotency check (BEFORE any network work):
  //     - Read existing publish.json via readPublishRecord(absFolder).
  //     - If existing && existing.videoId && !opts.force:
  //         return { skipped: true, reason: 'video-exists', publishRecordPath, record: existing }.
  //     - If opts.force: reason = 'force'; proceed.
  //     - Else: reason = 'first-publish'; proceed.
  //  7. Load OAuth token via loadToken(channel, { clientId: opts.clientId, clientSecret: opts.clientSecret }). Throws OAuthError on refresh failure (propagates with the run-auth remediation already shaped).
  //  8. Build UploadEpisodeArgs:
  //       { client, channel, manifest, episodePath: <abs .golazo/episode.mp4>, thumbnailPath: <abs .golazo/thumb.png>, clock: opts.clock }
  //  9. Call publishWithRetry(args, opts.retryOpts). Returns PublishRecord on success; throws QuotaExceededError / OAuthError / UploadError / GaxiosError on failure. Propagate all.
  // 10. Write publish.json via writePublishRecord(absFolder, record). The record's `privacyStatus` field comes directly from `publishWithRetry`'s return value, which uses `PRIVACY_STATUS` from `../publish/uploader.js`. No bare string is constructed at the orchestrator boundary.
  // 11. Return { skipped: false, reason, publishRecordPath, record }.
