  export const YOUTUBE_UPLOAD_SCOPE = 'https://www.googleapis.com/auth/youtube.upload';

  // googleapis 144.x exposes `google.auth.OAuth2` (constructor). For TS, import
  // `OAuth2Client` from 'google-auth-library' (which googleapis re-exports) for the type.
  import { google } from 'googleapis';
  import type { OAuth2Client, Credentials } from 'google-auth-library';

  export function createOAuth2Client(opts?: {
    clientId?: string;       // default: process.env.GOOGLE_CLIENT_ID
    clientSecret?: string;   // default: process.env.GOOGLE_CLIENT_SECRET
    redirectUri?: string;    // default: 'urn:ietf:wg:oauth:2.0:oob' (Out-Of-Band manual code flow)
  }): OAuth2Client;
  // Throws OAuthError if clientId/clientSecret are missing.

  export function buildAuthUrl(client: OAuth2Client): string;
  // generateAuthUrl({ access_type: 'offline', prompt: 'consent', scope: [YOUTUBE_UPLOAD_SCOPE] })
  // access_type=offline is REQUIRED for a refresh_token to be returned.
  // prompt=consent is REQUIRED so re-authorising the same Google account still returns a refresh_token.

  export async function exchangeCode(client: OAuth2Client, code: string): Promise<Credentials>;
  // Wraps client.getToken(code). Throws OAuthError({ field: 'exchange', ... }) on failure.
  // If returned credentials has no refresh_token: throws OAuthError({ field: 'refresh_token', reason: 'Google did not return a refresh_token; access_type=offline + prompt=consent required', remediation: 'rerun golazo auth <kid> and accept the consent screen' }).

  export function saveToken(tokenPath: string, credentials: Credentials): void;
  // Atomic write: writeFileSync(tokenPath, JSON.stringify(credentials, null, 2) + '\n', 'utf8').
  // Ensures parent directory exists via mkdirSync(..., { recursive: true }).
  // Sets file mode 0o600 (owner-read/write only) after write — these are personal credentials.
  // NEVER logs the credentials.

  export async function loadToken(channelConfig: ChannelConfig, opts?: {
    clientId?: string;
    clientSecret?: string;
  }): Promise<OAuth2Client>;
  // 1. existsSync(channelConfig.youtube.oauthTokenPath) — else throw OAuthError({ field: 'tokenPath', reason: `oauth token file not found at <path>`, remediation: `run 'golazo auth <kid>'` })
  // 2. Read + JSON.parse the file. On JSON.parse failure: throw OAuthError({ field: 'tokenJson', ... }).
  // 3. Create OAuth2Client; setCredentials(creds).
  // 4. Register a token-refresh listener: client.on('tokens', (newCreds) => { merge with stored refresh_token; saveToken(path, merged) }). Required because googleapis only emits the new access_token on the 'tokens' event when it auto-refreshes.
  // 5. If creds.expiry_date && creds.expiry_date < Date.now(): force a refresh now via client.refreshAccessToken() (deprecated but still functional in googleapis 144.x — alternative is to make any API call, which will auto-refresh). Wrap in try/catch; on refresh failure throw OAuthError({ field: 'refresh', reason: <err.message>, remediation: `run 'golazo auth <kid>' to reauthorize` }).
  // 6. Return the configured client. Downstream callers (Plans 03-03+) pass this client straight to `google.youtube({ version: 'v3', auth: client })`.

  export interface RunAuthOpts {
    kid: string;
    channelsPath?: string;        // default './channels.yaml'
    clientId?: string;            // default env
    clientSecret?: string;        // default env
    /**
     * Function that produces the authorization code from the user. Injected
     * so tests can stub it without touching stdin. The default reads one
     * line from process.stdin (after the CLI handler prints the consent URL).
     */
    readCode?: (authUrl: string) => Promise<string>;
  }

  export interface RunAuthResult {
    tokenPath: string;        // absolute, where the token was written
    channelId: string;        // for the success message
  }

  export async function runAuth(opts: RunAuthOpts): Promise<RunAuthResult>;
  // 1. Load channels.yaml via the new permissive loader (loadChannelsFile({ ..., skipTokenCheck: true })). Look up the entry for opts.kid; if missing throw UnknownKidError (reused from src/config/errors.ts).
  // 2. Resolve the absolute oauthTokenPath the same way loadChannelsFile does (tilde-expand against HOME, then resolve against channels.yaml's parent dir).
  // 3. Build the OAuth2Client + auth URL.
  // 4. Call opts.readCode(authUrl) to get the code from the user.
  // 5. exchangeCode(client, code).
  // 6. saveToken(tokenPath, credentials).
  // 7. Return { tokenPath, channelId }.
