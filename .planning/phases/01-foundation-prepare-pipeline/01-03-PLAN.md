  // src/prepare/types.ts
  export type MatchResult = 'W' | 'L' | 'D';
  export interface GameMeta {
    date: string;              // ISO date 'YYYY-MM-DD' (validated, no further normalization)
    opponent: string;          // slug as it appears in the folder name, e.g. 'united', 'city-sc', 'ac-milan'
    scoreFor: number;          // non-negative integer
    scoreAgainst: number;      // non-negative integer
    result: MatchResult;       // derived: scoreFor>scoreAgainst → 'W'; <  → 'L'; == → 'D'
  }

  // src/prepare/filename.ts
  export const FILENAME_REGEX: RegExp;
  // exactly /^(\d{4}-\d{2}-\d{2})_vs_([a-z0-9]+(?:-[a-z0-9]+)*)_(\d+)-(\d+)$/
  // - date is 4-2-2 digits (further validated by Date.parse equivalence — see action)
  // - opponent slug is one or more lowercase-alphanumeric groups separated by single hyphens (no leading/trailing hyphen, no consecutive hyphens)
  // - scoreFor, scoreAgainst are one or more digits, parsed as Number, must be 0..99

  export function parseFilename(folderName: string): GameMeta;
  // throws FilenameError when regex does not match OR date is not a real calendar date

  // src/prepare/kid.ts
  export function resolveKidFromPath(
    absoluteFolderPath: string,
    opts?: { channelsPath?: string }
  ): string;
  // - splits the absolute path into segments (path.sep), finds the segment immediately after the segment 'golazo'
  // - if no such segment exists, throws KidPathError with remediation pointing to '~/golazo/<kid>/<game>/' layout
  // - loads channels.yaml via loadChannelsFile, gets valid kid keys via Object.keys
  // - if the candidate kid is not in the valid keys, throws UnknownKidError (reusing the Plan 02 class)
  // - returns the kid key as a string
